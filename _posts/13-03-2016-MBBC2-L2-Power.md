#### R codes Using Monte Carlo

{% highlight r linenos %}

nosim <- 100000
n <- 16
sigma <- 4
mu0 <- 30
mua <- 32
z <- rnorm(nosim)
xsq <- rchisq(nosim, df = 15)
t <- qt(.95, 15)
mean(z + sqrt(n) * (mua - mu0) / sigma >
     t / sqrt(n - 1) * sqrt(xsq))
##result is 60%
